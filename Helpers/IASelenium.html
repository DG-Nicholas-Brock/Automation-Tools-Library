 26    def click_at_offset(self, x: int = 0, y: int = 0) -> None:
 27        """
 28        Move some number of pixels away from the current position of the cursor and then click.

 35    def click_drag_release(self, web_element: WebElement, x: int, y: int) -> None:
 36        """
 37        Click-and-hold on a WebElement, then drag it some number of pixels before finally releasing the held click.
 38
 39        :param web_element: The WebElement to be dragged.
 40        :param x: The number of pixels to drag left/right. Positive values are right, and negative values are left.
 41        :param y: The number of pixels to drag up/down. Positive values are down, and negative values are up.
 42        """
 43        ActionChains(self.driver) \
 44            .move_to_element(web_element) \
 45            .click_and_hold(on_element=web_element) \
 46            .move_by_offset(x, y) \
 47            .release() \
 48            .perform()

 81        """
 82        Obtain the zero-based index of the tab currently in use by the browser.
 83
 84        :returns: The index of the tab currently in use by the browser.

 86        handle = self.driver.current_window_handle

 88        for i in range(count):

 90                return i
 91        assert False, 'We never found a matching window handle.'

255    def set_inner_window_dimensions(self, inner_width=None, inner_height=None) -> None:
256        """
257        Set the dimensions of the inner viewport of the browser.
258
259        :param inner_width: The desired width of the viewport.
260        :param inner_height: The desired height of the viewport.
261
262        :raises ValueError: If the supplied inner_width is less than 500px.
263        """
264        if inner_width is not None and inner_width < 500:
265            raise ValueError("Browser constraints prevent setting of inner width below 500px.")
266        current_size = self.driver.get_window_size()
267        current_width = current_size['width']  # OUTER dimension
268        current_height = current_size['height']  # OUTER dimension
269        adjusted_width = current_width
270        adjusted_height = current_height
271        if inner_width:
272            adjusted_width = current_size['width'] + inner_width - self.get_inner_width()
273        if inner_height:
274            adjusted_height = current_size['height'] + inner_height - self.get_inner_height()
275        print(f'\nAdjusting browser outer dimensions to {adjusted_width}x{adjusted_height}')
276        # this actually sets the OUTER dimensions to our adjusted dimensions
277        self.driver.set_window_size(width=adjusted_width, height=adjusted_height)

290    def switch_to_tab_by_index(self, zero_based_index: int) -> None:
291        """
292        Switch to a specific tab based on its zero-based index.
293
294        :param zero_based_index: The index of the tab to switch to.
295
296        :raises TimeoutException: If there are fewer open tabs than the supplied index.
297        """
298        # if targeting anything beyond the first tab, wait for it to exist / finish opening
299        if zero_based_index > 0:
300            necessary_length = zero_based_index + 1
301
302            def desired_tab_exists() -> bool:
303                handles = len(self.driver.window_handles)
304                return handles >= necessary_length
305            try:
306                WebDriverWait(self.driver, 1).until(IAec.function_returns_true(
307                    custom_function=desired_tab_exists,
308                    function_args={}))
309            except TimeoutException as toe:
310                raise TimeoutException(
311                    msg=f"An attempt was made to switch tabs to index {zero_based_index}, but there weren't that "
312                        f"many tabs present.") from toe
313        handle = self.driver.window_handles[zero_based_index]
314        self.switch_to_tab_by_title(title=handle)
315        time.sleep(1)
316
317    def switch_to_tab_by_title(self, title: str) -> None:
318        """
319        Switch to a tab by the text of the tab.
320
321        :param title: The text content of the tab to switch to.
322        """
323        assert title in self.driver.window_handles, f'There is no tab titled \'{title}\' in the available tab listings.'
324        self.driver.switch_to.window(title)
325        time.sleep(1)

327    def take_screenshot(self, directory: str = 'Screenshots', screenshot_name: str = None) -> str:
328        """
329        Take a screenshot of the tab currently in use.
330
331        :param directory: The path of the directory into which the file will be saved.
332        :param screenshot_name: The name to supply to the screenshot. If no name is provided, a filename which includes
333            the current time will be generated.
334
335        :returns: The location and name of the saved screenshot.
336        """
337        current_time = str(datetime.datetime.now()).replace(' ', '').replace(':', '')
338        screenshot_title = 'AutoScreenshot' + current_time
339        if screenshot_name:
340            screenshot_title = str(screenshot_name)
341        screenshot_title += '.png'
342        screenshot_folder = f'{os.getcwd()}/{directory}'
343        if not os.path.exists(screenshot_folder):
344            os.mkdir(screenshot_folder)
345        safe_path = FilePathFormatting.system_safe_file_path(screenshot_folder, screenshot_title)
346        self.driver.save_screenshot(safe_path)
347        self.logger.info(msg=f'Screenshot taken: {safe_path}')
348        return safe_path
