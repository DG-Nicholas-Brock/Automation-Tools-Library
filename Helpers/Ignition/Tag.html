116    def to_dict(self) -> Dict:
117        local_tag = self.duplicate()
118        special_case = ['_name']
119        not_used = ['_path', '_base_path', '_provider', '_full_path', '_type_id_path']
120        # Get set attributes of the Tag Object
121        attrs = vars(
122            local_tag)
123        tag_dict = {}
124        for item in not_used:
125            if item in attrs:
126                del attrs[item]
127        for item in special_case:
128            new_item = item.replace('_', '')
129            attrs[new_item] = attrs[item]
130            del attrs[item]
131        for attr in attrs:
132            ignition_key = self._to_camel_case(
133                name=attr)
134            if getattr(local_tag, attr) is None:
135                pass
136            elif isinstance(getattr(local_tag, attr), Enum):
137                tag_dict[ignition_key] = getattr(local_tag, attr).value
138            else:
139                tag_dict[ignition_key] = getattr(local_tag, attr)

156    def diff_key_values(self, expected_tag):
157        """Compares the keys 2 tag definitions have in common and returns value differences"""
158        diff_values = {}
159        for my_key in self.to_dict().keys():
160            if my_key not in self.diff_my_unmatched_keys(
161                    compare_tag=expected_tag):
162                if self.to_dict()[my_key] != expected_tag.to_dict()[my_key]:
163                    diff_values[my_key] = {'actual': self.to_dict()[my_key], 'expected': expected_tag.to_dict()[my_key]}
164        return diff_values

167        """Compares against another tag definition and returns keys & values that are unique to this tag"""
168        result = {}
169        for key in self.diff_my_unmatched_keys(

183        """Compares against another tag definition and returns keys that are unique to this tag"""

192            dict_keys2=self.to_dict().keys())

330    @staticmethod
331    def get_tag_differences(actual: list, expected: list):
332        """Get the differences between two list of string representations of dicts that contain Tag Definitions
333        This is normally what is returned from an Ignition Gateway"""
334        actual_tags = []
335        expected_tags = []
336        tags_processed = []
337        failures = {}
338        for tag in actual:
339            actual_tags.append(TagHelper.build_tag_from_string(
340                dict_as_string=json.dumps(tag)))
341        for tag in expected:
342            expected_tags.append(TagHelper.build_tag_from_string(
343                dict_as_string=json.dumps(tag)))
344
345        for expected_tag in expected_tags:
346            for actual_tag in actual_tags:
347                if actual_tag.get_full_path() == expected_tag.get_full_path():
348                    tags_processed.append(expected_tag.get_full_path())
349                    if expected_tag.to_dict() != actual_tag.to_dict():
350                        failures[expected_tag.get_full_path()] = actual_tag.diff(expected_tag=expected_tag)
351        for expected_tag in expected_tags:
352            if expected_tag.get_full_path() not in tags_processed:
353                failures[expected_tag.get_full_path()] = {'actual': '', 'expected_tag': expected_tag.to_dict()}
354        for actual_tag in actual_tags:
355            if actual_tag.get_full_path() not in tags_processed:
356                failures[actual_tag.get_full_path()] = {'actual': actual_tag.to_dict(), 'expected_tag': ''}
357
358        return failures

402    @staticmethod
403    def build_tag_from_dictionary(tag_configs: Dict):
404        provider = ''
405        path = ''
406        name = tag_configs.get('name')
407        if 'path' in tag_configs:
408            provider = f'[{tag_configs.get("path").get("source")}]'
409            path = TagHelper._process_path(tag_configs.get('path', '').get('pathParts', ''))
410        match tag_configs.get('tagType'):
411            case 'UdtInstance':
412                tag = UdtInstance(name=name, provider=provider, path=path, type_id='')
413            case 'UdtType':
414                tag = UdtDef(name=name, provider=provider, path=path)
415            case 'Folder':
416                tag = Folder(name=name, provider=provider, path=path)
417            case _:
418                tag = Tag(name=name, provider=provider, path=path)
419        for key in tag_configs.keys():
420            tag = TagHelper._process_key(tag, key, tag_configs[key])
421        return tag
422
423    @staticmethod
424    def build_tag_from_string(dict_as_string: str):
425        """Build a Tag, UdtInstance, UdtDef or Folder object from a string representation of a dict"""
426        provider = ''
427        path = ''
428        response_json = json.loads(dict_as_string)
429        if 'error' not in response_json:
430            if isinstance(response_json, list):
431                response_json = response_json[0]
432            name = response_json.get('name')
433            if 'path' in response_json:
434                provider = f'[{response_json.get("path").get("source")}]'
435                path = TagHelper._process_path(response_json.get('path', '').get('pathParts', ''))
436            if response_json.get('tagType') == 'UdtInstance':
437                tag = UdtInstance(name=name, provider=provider, path=path, type_id='')
438            elif response_json.get('tagType') == 'UdtType':
439                tag = UdtDef(name=name, provider=provider, path=path)
440            elif response_json.get('tagType') == 'Folder':
441                tag = Folder(name=name, provider=provider, path=path)
442            else:
443                tag = Tag(name=name, provider=provider, path=path)
444            for key in response_json.keys():
445                tag = TagHelper._process_key(tag, key, response_json[key])
446            return tag

456    @staticmethod
457    def _process_path(path_parts: json):
458        path_parts.pop()

461    @staticmethod
462    def _process_key(tag: Tag, key: str, value: object):
463        excluded_types = ['path', 'name']
464        if key not in excluded_types:
465            mapped_key = TagHelper._to_snake_case(key)
466            setattr(tag, mapped_key, value)
467        return tag

469    @staticmethod
470    def _to_snake_case(name):
471        name = re.sub('(.)([A-Z][a-z]+)', r'\1_\2', name)
472        name = re.sub('__([A-Z])', r'_\1', name)
473        name = re.sub('([a-z0-9])([A-Z])', r'\1_\2', name)
474        return name.lower()
