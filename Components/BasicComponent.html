 71        """
 72        Click the component.
 73
 74        :param wait_timeout: Poll the DOM up to this amount of time before potentially throwing a TimeoutException.
 75            Overrides the default of the component.
 76        :param binding_wait_time: The amount of time to wait after the click event occurs before continuing.
 77
 78        :raises TimeoutException: If the component is not found in the DOM.
 79        """
 80        wait = WebDriverWait(driver=self.driver, timeout=wait_timeout) if wait_timeout is not None else self.wait

 84        wait.until(
 85            IAec.function_returns_true(custom_function=self._click, function_args={}),
 86            message="The element never became clickable.")
 87        self.wait_on_binding(time_to_wait=binding_wait_time)

120    def find(self, wait_timeout: Optional[Union[int, float]] = None) -> WebElement:
121        """
122        Get a singular Web Element in the DOM which matches the supplied locator for this component. If more than one
123        Web Element would match, return only the first match encountered in the DOM from top to bottom in the document
124        - NOT the viewable area. Distinct from `xfind`, which uses XPATH.
125
126        :param wait_timeout: The amount of time (in seconds) to wait to locate the Web Element.
127
128        :returns: The first Web Element found given the supplied locator for this component.
129
130        :raises TimeoutException: If no element matches the locator of this component.
131        """
132        if wait_timeout is not None:  # 0 is acceptable
133            local_wait = WebDriverWait(
134                self.driver,
135                wait_timeout)
136        else:
137            local_wait = self.wait
138        css_locator = self.get_full_css_locator()
139        try:
140            return local_wait.until(
141                ec.presence_of_element_located(
142                    css_locator))
143        except TimeoutException as toe:
144            description = f"\nDescription: {self.description}" if self.description else ''
145            raise TimeoutException(
146                msg=f"Unable to locate element with CSS locator: {css_locator}{description}") from toe

317    def wait_on_text_condition(
318            self,
319            text_to_compare: Optional[Any],
320            condition: Union[IAec.TextCondition, IAec.NumericCondition],
321            wait_timeout: Optional[float] = None) -> str:
322        """
323        Obtain the text of this Component, after potentially waiting some period of time for the Component to display
324        that text.
325
326        :param text_to_compare: The text to be used in conjunction with the supplied condition. If the supplied value
327            is None, then no wait will ever occur and the text of the component will be immediately returned.
328        :param condition: The condition to be used to compare the Component text to the provided text.
329        :param wait_timeout: The amount of time (in seconds) you are willing to wait for the Component to display
330            the specified text. If not supplied, this will default to the wait timeout supplied in the constructor
331            of the Component.
332
333        :returns: The text of the Component, after potentially having waited for an expected text match.
334        """
335        cond_wait = WebDriverWait(driver=self.driver, timeout=wait_timeout) if wait_timeout is not None else self.wait
336        text_to_compare = str(text_to_compare) if text_to_compare is not None else text_to_compare
337        text = ""
338
339        def compare_against_condition():
340            """
341            Comparisons are only completed when the supplied comparison value is not None. One check is always executed
342            to safeguard against StaleElementReferenceExceptions.
343
344            :return: True, if the text of the Component satisfies the specified condition - False otherwise.
345
346            :raises NotImplementError: If the specified condition is not supported.
347            """
348            nonlocal text
349            nonlocal text_to_compare
350            try:
351                text = self.get_text()
352            except TimeoutException as get_text_toe:
353                get_text_toe.should_raise = True  # toe from failing to locate the component/element should be raised
354                raise
355            if text_to_compare is None:
356                # immediately return because the text will NEVER compare correctly against None
357                return text
358            if condition == IAec.TextCondition.EQUALS:
359                return text_to_compare == text
360            elif condition == IAec.TextCondition.DOES_NOT_EQUAL:
361                return text_to_compare != text
362            elif condition == IAec.TextCondition.CONTAINS:
363                return text_to_compare in text
364            elif condition == IAec.TextCondition.DOES_NOT_CONTAIN:
365                return text_to_compare not in text
366            elif condition == IAec.NumericCondition.EQUALS:
367                return str(text_to_compare).replace(",", "") == str(text).replace(",", "")
368            elif condition == IAec.NumericCondition.DOES_NOT_EQUAL:
369                return str(text_to_compare).replace(",", "") != str(text).replace(",", "")
370            else:
371                raise NotImplementedError("Unhandled condition while comparing text.")
372
373        try:
374            cond_wait.until(IAec.function_returns_true(custom_function=compare_against_condition, function_args={}))
375        except TimeoutException as toe:
376            if getattr(toe, 'should_raise', False):
377                raise  # due to failing to find the component/element
378            pass  # otherwise pass because toe originated from not meeting condition
379        return text

781    @staticmethod
782    def get_css_locator(locators: list) -> Tuple[Union[By, str], str]:
783        css_list = []
784        for locator in locators:
785            by = By.CSS_SELECTOR
786            value = locator
787            if type(locator) == tuple:
788                by = locator[0]
789                value = locator[1]
790            if by == By.XPATH:
791                raise TypeError(
792                    f'Locator: {locator} has a By type of XPATH which is incompatible with the css_selector builder.')
793            elif by == By.ID:
794                css_list.append(f'[id="{value}"]')
795            elif by == By.CLASS_NAME:
796                css_list.append(f'.{value}')
797            elif by == By.NAME:
798                css_list.append(f'[name="{value}"]')
799            else:
800                css_list.append(value)
801        return By.CSS_SELECTOR, " ".join(css_list)
802
803    @staticmethod
804    def get_xpath_locator(locators: list) -> Tuple[Union[By, str], str]:
805        xpath_list = []
806        for locator in locators:
807            by = locator[0]
808            value = locator[1]
809            if by == By.CSS_SELECTOR:
810                raise TypeError(
811                    f'Locator: {locator} has a By type of CSS_SELECTOR which is incompatible with the xpath builder.')
812            elif by == By.ID:
813                xpath_list.append(f'//*[@id="{value}"]')
814            elif by == By.CLASS_NAME:
815                xpath_list.append(f'//*[contains(concat(" ", normalize-space(@class), " "),"{value}")]')
816            elif by == By.NAME:
817                xpath_list.append(f'//*[@name="{value}"]')
818            elif by == By.LINK_TEXT:
819                xpath_list.append(f'//a[normalize-space()="{value}"]')
820            elif by == By.PARTIAL_LINK_TEXT:
821                xpath_list.append(f'//a[contains(normalize-space(),"{value}")]')
822            elif by == By.TAG_NAME:
823                xpath_list.append(f'//{value}')
824            else:
825                xpath_list.append(value)
826        return By.XPATH, "".join(xpath_list)
