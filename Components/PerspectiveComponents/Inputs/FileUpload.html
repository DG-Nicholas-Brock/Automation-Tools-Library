134    def all_extensions_allowed(self) -> bool:
135        """
136        Determine if all extensions are currently allowed for the File Upload. Does not work for File Uploads which use
137        the "small" layout, because we would need to expand the modal - but there is no good way to then remove the
138        modal.
139
140        :returns: True, if there are no file extension restrictions in place for the component. False if any
141            restrictions are in place.
142
143        :raises TypeError: If the File Upload is in the "small" layout. This error is raised because there could
144            actually be restrictions in place, but we are unable to locate them.
145        """
146        if self.is_using_small_layout():
147            raise ValueError("Unable to verify file restrictions while the File Upload is using the 'small' layout.")
148        return not self.is_displaying_hover_info_icon()

150    def clear_uploads(self, max_number_of_attempts: int = 10) -> None:
151        """
152        Reset the File Upload Component back to its default state after having uploaded files.
153
154        :param max_number_of_attempts: How many times you are willing to click the next page arrow in order to reach
155            the upload summary before giving up.
156
157        :raises TimeoutException: If there were no uploads to clear, or if we failed to reach the summary of uploads
158            after clicking next page arrow as many times as specified in max_number_of_attempts.
159        """
160        attempts = 0
161        while not self.summary_panel_is_visible() and attempts < max_number_of_attempts:
162            self.click_next_page()
163            attempts += 1
164        self._get_clear_uploads_link().click(wait_timeout=1)

190    def extension_is_allowed(self, extension: str) -> bool:
191        """
192        Determine if the supplied case-insensitive extension is listed as being restricted.
193
194        :returns: True, if the supplied case-insensitive extension is currently restricted - False otherwise.
195        """
196        all_extensions_allowed = self.all_extensions_allowed()
197        if not all_extensions_allowed:
198            extension = extension.replace(".", "").upper()
199            if self.is_displaying_hover_info_icon():
200                return extension in self.get_information_hover_text()
201            else:
202                return extension in self._secondary_message.get_text()
203        return all_extensions_allowed

268    def get_total_count_of_results(self) -> int:
269        """
270        Obtain a count of how many files were uploaded during the last event.
271
272        :raises TimeoutException: If the component is not currently displaying a summary of the upload.
273        """
274        on_summary_page = self.get_current_page_index() == -1
275        if on_summary_page:
276            self.click_previous_page()
277            self.wait_on_binding(time_to_wait=0.5)
278        count_of_results = int(self._file_message.get_text().split(" ")[-1])
279        if on_summary_page:
280            self.click_next_page()
281        return count_of_results

334    def navigate_to_file_summary_by_index(self, desired_index: int) -> None:
335        """
336        Within the upload summary pagination area, navigate to a specific page index. To navigate to the Summary panel,
337        supply a value of -1.
338
339        :raises TimeoutException: If the pagination panel is not currently displayed.
340        """
341        if self.summary_panel_is_visible():
342            self.click_previous_page()
343        current_index = self.get_current_page_index()
344        # stop if we reach the desired page or the summary page
345        while current_index != desired_index and current_index != -1:
346            self._previous_page_link.click(binding_wait_time=0.5) if current_index > desired_index \
347                else self._next_page_link.click(binding_wait_time=0.5)
348            current_index = self.get_current_page_index()

392    def upload_files(self, normalized_file_path_list: List[str]) -> None:
393        """
394        Upload a list of files by supplying normalized file paths.
395
396        :param normalized_file_path_list: A list of OS-agnostic string file paths.
397        """
398        self.driver.execute_script('arguments[0].style.display="block";', self._find_file_input())
399        self._find_file_input().send_keys('\n'.join(normalized_file_path_list))
400        self.driver.execute_script('arguments[0].style.display="none";', self._find_file_input())
