307    def select_date(self, date: PerspectiveDate, collapse_when_done: bool = True) -> None:
308        """
309        Select a date within the DateTime Input
310
311        :param date: The date to select.
312        :param collapse_when_done: If True, the picker will be collapsed when done. If False, the picker will be left
313            expanded.
314        """
315        self.expand()
316        self._picker.select_date_only_not_time(date=date)
317        if date.hours12 or date.hours24 or date.minutes or date.seconds:
318            # re-expand after potentially selecting a day
319            self.expand()
320            self._picker.select_time_only_not_date(date=date)
321            if not self.is_expanded():
322                self.click()
323            self._picker.select_date_only_not_time(date=PerspectiveDate(day=date.day))
324        if collapse_when_done:
325            self._collapse()
