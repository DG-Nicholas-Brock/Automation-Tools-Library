 15class PerspectiveDate:
 16    """
 17    Perspective inputs use a unique take on dates, and formatting applied to these inputs can result in some values
 18    making no sense. Is a supplied time of 11:12:13 AM or PM? Should users specify a month by name or index? If index,
 19    is the month zero-indexed or one-indexed? By forcing interaction with these components to use a uniform class, we
 20    alleviate many pain points and a lot of confusion.
 21    """
 22
 23    def __init__(
 24            self,
 25            year: Optional[int] = None,
 26            month: Optional[Union[int, str]] = None,
 27            day: Optional[int] = None,
 28            hours24: Optional[int] = None,
 29            minutes: Optional[int] = None,
 30            seconds: Optional[int] = None):
 31        """
 32        Create a date-type object Perspective can actually do something with.
 33        :param year: The numeric year of this date.
 34        :param month: accepts int or str (1, or "January"), but will be coerced into an int.
 35        :param day: The numeric day you'd like to select.
 36        :param hours24: The hours you'd like to select in 24-hour time. The am/pm value will be derived from this
 37            value, as will the hours12 value.
 38        :param minutes: The numeric minute of this date/time object.
 39        :param seconds: The numeric seconds of this date/time object.
 40        """
 41        self.year = year
 42        self._set_month(month=month)  # accept int/str but refine to int
 43        self.day = day
 44        self.hours24 = hours24 % 24 if hours24 else hours24
 45        self._set_hours12()  # calculation to be made
 46        self._set_am_pm()  # calculation to be made
 47        self.minutes = minutes
 48        self.seconds = seconds

 50    def __str__(self) -> str:
 51        """Provide a human-readable string representation of this object. Fill missing pieces with "X"."""
 52        filler = "X"
 53        return f"{self.month or filler}/{self.day or filler}/{self.year or filler} " \
 54               f"{self.hours24 or 2 * filler}:{self.minutes or 2 * filler}:{self.seconds or 2 * filler} (24-hour time)"

 56    def get_new_date_with_offset(
 57            self,
 58            year_offset: Optional[int] = None,
 59            month_offset: Optional[int] = None,
 60            day_offset: Optional[int] = None,
 61            hours24_offset: Optional[int] = None,
 62            minutes_offset: Optional[int] = None,
 63            seconds_offset: Optional[int] = None):
 64        """
 65        Obtain a new date with aspects of the current ('base') date modified by some value.
 66
 67        :param year_offset: The count of years to add to the base date.
 68        :param month_offset: The count of months to add to the base date.
 69        :param day_offset: The count of days to add to the base date.
 70        :param hours24_offset: The count of hours to add to the base date. Must be less than 24. If larger than 24,
 71            combine with day_offset.
 72        :param minutes_offset: The count of minutes to add to the base date. Must be between -60 and 60. If the absolute
 73            value would be larger than 60, combine with hours24_offset.
 74        :param seconds_offset: The count of seconds to add to the base date. Must be between -60 and 60. If the absolute
 75            value would be larger than 60, combine with minutes_offset.
 76
 77        :returns: A new Perspective Date object offset by some values from a base Perspective Date.
 78
 79        :raises AssertionError: if the 'base' date being modified does not contain a value for some aspect which has
 80            been specified for modification, or if day offset values which would result in a day of a new month are
 81            supplied.
 82        """
 83        new_year = self.year
 84        new_month = self.month
 85        new_day = self.day
 86        new_hours24 = self.hours24
 87        new_minutes = self.minutes
 88        new_seconds = self.seconds
 89        if year_offset:
 90            IAAssert.is_true(
 91                value=self.year is not None,
 92                failure_msg="The supplied date had no year to offset from.")
 93            new_year = self.year + year_offset
 94        if month_offset:
 95            IAAssert.is_true(
 96                value=self.month is not None,
 97                failure_msg="The supplied date had no month to offset from.")
 98            new_month = self.month + month_offset
 99            if new_month < 1:
100                new_month = 12 + (new_month % 12)
101        if day_offset:
102            IAAssert.is_true(
103                value=self.day is not None,
104                failure_msg="The supplied date had no day to offset from.")
105            IAAssert.is_greater_than(
106                left_value=day_offset + self.day,
107                right_value=0,
108                failure_msg="Unable to calculate day with negative offset which would result in new day being "
109                            "calculated from end of month because we have no way of knowing what month is being "
110                            "interacted with.")
111            IAAssert.is_less_than_or_equal_to(
112                left_value=day_offset + self.day,
113                right_value=31,
114                failure_msg="There can only be up to 31 days in a month.")
115            new_day = self.day + day_offset
116            # no special handling for new day because assertions prevent next/previous month checking
117        if hours24_offset:
118            new_hours24 = self.hours24 + hours24_offset
119            if new_hours24 < 0:
120                new_hours24 = 24 - (new_hours24 % 24)
121            IAAssert.is_true(
122                value=self.hours24 is not None,
123                failure_msg="The supplied date had no hours to offset from.")
124            IAAssert.is_less_than(
125                left_value=abs(hours24_offset),
126                right_value=24,
127                failure_msg="There are only 0-23:59 hours in a day; please modify the day offset for offsets greater "
128                            "than 24 hours.")
129        if minutes_offset:
130            IAAssert.is_true(
131                value=self.minutes is not None,
132                failure_msg="The supplied date had no minutes to offset from.")
133            IAAssert.is_less_than(
134                left_value=abs(minutes_offset),
135                right_value=60,
136                failure_msg="There are only 0-59 minutes in an hour; please modify the hour offset instead for "
137                            "offsets greater than 60 minutes.")
138            new_minutes = self.minutes + minutes_offset
139            if new_minutes < 0:
140                new_minutes = 60 - (new_minutes % 60)
141        if seconds_offset:
142            IAAssert.is_true(
143                value=self.seconds is not None,
144                failure_msg="The supplied date had no seconds to offset from.")
145            IAAssert.is_less_than(
146                left_value=abs(seconds_offset),
147                right_value=60,
148                failure_msg="There are only 0-59 seconds in a minute; please modify the minute offset instead for "
149                            "offsets greater than 60 seconds.")
150            new_seconds = self.seconds + seconds_offset
151            if new_seconds < 0:
152                new_seconds = 60 - (new_seconds % 60)
153        return PerspectiveDate(
154            year=new_year,
155            month=new_month,
156            day=new_day,
157            hours24=new_hours24,
158            minutes=new_minutes,
159            seconds=new_seconds
160        )

162    def _set_am_pm(self) -> None:
163        """
164        Apply the 12 hour meridiem to use based on the 24-hour supplied value.
165        """
166        if self.hours24:
167            self.am_pm = "am" if self.hours24 < 12 else "pm"
168        else:
169            self.am_pm = None
170
171    def _set_hours12(self) -> None:
172        """
173        Apply a 12-hour (am/pm) value for instances where we might need to use a 12-hour format.
174        """
175        if self.hours24 is not None:
176            self.hours12 = self.hours24 - 12 if self.hours24 > 12 else self.hours24
177        else:
178            self.hours12 = None

180    def _set_month(self, month: Union[int, str]) -> None:
181        """
182        Determine and apply a numeric month based on whether the user supplied the month as a number or a string.
183        """
184        try:
185            self.month = int(month)
186        except ValueError:
187            # string
188            month_dict = {v: k for k, v in enumerate(calendar.month_name)}
189            self.month = month_dict[month]
190        except TypeError:
191            # None
192            self.month = month

495    def select_month(self, month: Union[int, str]) -> None:
496        """
497        Select a month by using the chevrons/arrows. Selenium has issues with the dropdowns when they are opened in
498        Popups, so we avoid their use at all.
499
500        :param month: The month to select. This may be a numeric representation or the string name of the month.
501
502        :raises AssertionError: If the picker could not apply the month.
503        """
504        try:
505            month = int(month)
506        except ValueError:
507            month_dict = {v: k for k, v in enumerate(calendar.month_name)}
508            month = month_dict[month]
509        selected_month_value = self._month_value()
510        count = 0
511        while selected_month_value != month and count < 12:
512            self._next_month_chevron.find().click() if selected_month_value < month \
513                else self._previous_month_chevron.find().click()
514            selected_month_value = self._month_value()
515            count += 1
516        IAAssert.is_equal_to(
517            actual_value=selected_month_value,
518            expected_value=month,
519            as_type=str,
520            failure_msg=f"Failed to set the month to a different 1-indexed value ({month}).")
521
522    def select_time_only_not_date(self, date: PerspectiveDate) -> None:
523        """
524        Select only a time (hours, minutes, seconds), while ignoring any date values.
525
526        :param date: The date to select, as an object. Only the time piece of the date object will be used.
527
528        :raises AssertionError: If unable to select the supplied time.
529        """
530        # seconds
531        seconds = date.seconds
532        if seconds is not None:
533            current_seconds = int(self.get_seconds())
534            if self._seconds_input.find().is_enabled():
535                target = self.previous_minute_chevron if int(seconds) < int(current_seconds) else \
536                    self.next_second_chevron
537                while current_seconds != int(seconds):
538                    target.click(binding_wait_time=0.5)
539                    new_seconds = int(self.get_seconds())
540                    if new_seconds != current_seconds:
541                        current_seconds = new_seconds
542                    else:
543                        break
544            IAAssert.is_equal_to(
545                actual_value=current_seconds,
546                expected_value=seconds,
547                as_type=int,

497        """
498        Select a month by using the chevrons/arrows. Selenium has issues with the dropdowns when they are opened in
499        Popups, so we avoid their use at all.
500
501        :param month: The month to select. This may be a numeric representation or the string name of the month.
502
503        :raises AssertionError: If the picker could not apply the month.
504        """
505        try:
506            month = int(month)
507        except ValueError:
508            month_dict = {v: k for k, v in enumerate(calendar.month_name)}
509            month = month_dict[month]
510        selected_month_value = self._month_value()
511        count = 0
512        while selected_month_value != month and count < 12:
513            self._next_month_chevron.find().click() if selected_month_value < month \
514                else self._previous_month_chevron.find().click()
515            selected_month_value = self._month_value()
516            count += 1
517        IAAssert.is_equal_to(
518            actual_value=selected_month_value,
519            expected_value=month,
520            as_type=str,
521            failure_msg=f"Failed to set the month to a different 1-indexed value ({month}).")
