 371                    ending_day=ending_day,

 373                    ending_year=year,

 415        def set_end_time(self, date: PerspectiveDate) -> None:
 416            """
 417            Set the ending time to use for the Historical range.
 418
 419            :param date: A PerspectiveDate object from which we will only use the time aspects.
 420
 421            :raises AssertionError: If any of the end time values do not match what is supplied after attempting to
 422                apply them.
 423            """
 424            am_pm_found = False
 425            if date.hours24 and self._end_time_hour_input.find().is_enabled():
 426                try:
 427                    # this picker determines our input approach (12 vs 24 hour)
 428                    am_pm_found = self._end_time_am_pm_select.find() is not None
 429                    target_value = date.hours12
 430                except TimeoutException:
 431                    # 24-hour format
 432                    am_pm_found = False
 433                    target_value = date.hours24
 434                self._set_end_hours(target_value=target_value)
 435            if date.minutes and self._end_time_minute_input.find().is_enabled():
 436                self._set_end_minutes(target_value=date.minutes)
 437            if am_pm_found:
 438                Select(webelement=self._end_time_am_pm_select.find()).select_by_value(value=date.am_pm)
 439            # assert new settings were applied
 440            if date.hours24 is not None:
 441                IAAssert.is_equal_to(
 442                    actual_value=self._end_time_hour_input.find().get_attribute("value"),
 443                    expected_value=date.hours12 if am_pm_found else date.hours24,
 444                    as_type=int,
 445                    failure_msg="Failed to set the hours for the end time.")
 446            if date.minutes is not None:
 447                IAAssert.is_equal_to(
 448                    actual_value=self._end_time_minute_input.find().get_attribute("value"),
 449                    expected_value=date.minutes,
 450                    as_type=int,
 451                    failure_msg="Failed to set the minutes for the end time.")
 452            if date.am_pm and date.am_pm is not None:
 453                IAAssert.is_equal_to(
 454                    actual_value=Select(
 455                        webelement=self._end_time_am_pm_select.find()).first_selected_option.text.upper(),
 456                    expected_value=date.am_pm.upper(),
 457                    as_type=str,
 458                    failure_msg="Failed to set AM/PM for the end time.")

 460        def set_range(self, historical_range: HistoricalRange) -> None:
 461            """
 462            Set the Historical range by specifying all time pieces of both the beginning and end dates.
 463
 464            :param historical_range: A HistoricalRange object containing information about both the start date/time and
 465                end date/time for the desired range.
 466
 467            :raises AssertionError: If unsuccessful in applying any of the range values.
 468            """
 469            _beginning_hours = 0
 470            _ending_hours = 0
 471            if historical_range.beginning_hours is not None:
 472                _beginning_hours = int(historical_range.beginning_hours)
 473            if not historical_range.beginning_am:
 474                _beginning_hours += 12
 475            if historical_range.ending_hours is not None:
 476                _ending_hours = int(historical_range.ending_hours)
 477            if not historical_range.ending_am:
 478                _ending_hours += 12
 479            beginning_date = PerspectiveDate(
 480                year=historical_range.beginning_year,
 481                month=historical_range.beginning_month,
 482                day=historical_range.beginning_day,
 483                hours24=_beginning_hours,

 496        def set_start_time(self, date: PerspectiveDate) -> None:
 497            """
 498            Set the starting time to use for the Historical range.
 499
 500            :param date: A PerspectiveDate object from which we will only use the time aspects.
 501
 502            :raises AssertionError: If any of the end time values do not match what is supplied after attempting to
 503                apply them.
 504            """
 505            am_pm_found = False
 506            if date.hours24 and self._start_time_hour_input.find().is_enabled():
 507                try:
 508                    # this picker determines our input approach (12 vs 24 hour)
 509                    am_pm_found = self._start_time_am_pm_select.find() is not None
 510                    target_value = date.hours12
 511                except TimeoutException:
 512                    # 24-hour format
 513                    am_pm_found = False
 514                    target_value = date.hours24
 515                self._set_start_hours(target_value=target_value)
 516            if date.minutes and self._start_time_minute_input.find().is_enabled():
 517                self._set_start_minutes(target_value=date.minutes)
 518            if am_pm_found:
 519                Select(webelement=self._start_time_am_pm_select.find()).select_by_value(value=date.am_pm)
 520            # assert new settings were applied
 521            if date.hours24 is not None:
 522                IAAssert.is_equal_to(
 523                    actual_value=self._start_time_hour_input.find().get_attribute("value"),
 524                    expected_value=date.hours12 if am_pm_found else date.hours24,
 525                    as_type=int,
 526                    failure_msg="Failed to set the hours for the start time.")
 527            if date.minutes is not None:
 528                IAAssert.is_equal_to(
 529                    actual_value=self._start_time_minute_input.find().get_attribute("value"),
 530                    expected_value=date.minutes,
 531                    as_type=int,
 532                    failure_msg="Failed to set the minutes for the start time.")
 533            if am_pm_found and date.am_pm is not None:
 534                IAAssert.is_equal_to(
 535                    actual_value=Select(
 536                        webelement=self._start_time_am_pm_select.find()).first_selected_option.text.upper(),
 537                    expected_value=date.am_pm.upper(),
 538                    as_type=str,
 539                    failure_msg="Failed to set AM/PM for the start time.")

 416        def set_end_time(self, date: PerspectiveDate) -> None:
 417            """
 418            Set the ending time to use for the Historical range.
 419
 420            :param date: A PerspectiveDate object from which we will only use the time aspects.
 421
 422            :raises AssertionError: If any of the end time values do not match what is supplied after attempting to
 423                apply them.
 424            """
 425            am_pm_found = False
 426            if date.hours24 and self._end_time_hour_input.find().is_enabled():
 427                try:
 428                    # this picker determines our input approach (12 vs 24 hour)
 429                    am_pm_found = self._end_time_am_pm_select.find() is not None
 430                    target_value = date.hours12
 431                except TimeoutException:
 432                    # 24-hour format
 433                    am_pm_found = False
 434                    target_value = date.hours24
 435                self._set_end_hours(target_value=target_value)
 436            if date.minutes and self._end_time_minute_input.find().is_enabled():
 437                self._set_end_minutes(target_value=date.minutes)
 438            if am_pm_found:
 439                Select(webelement=self._end_time_am_pm_select.find()).select_by_value(value=date.am_pm)
 440            # assert new settings were applied
 441            if date.hours24 is not None:
 442                IAAssert.is_equal_to(
 443                    actual_value=self._end_time_hour_input.find().get_attribute("value"),
 444                    expected_value=date.hours12 if am_pm_found else date.hours24,
 445                    as_type=int,
 446                    failure_msg="Failed to set the hours for the end time.")
 447            if date.minutes is not None:
 448                IAAssert.is_equal_to(
 449                    actual_value=self._end_time_minute_input.find().get_attribute("value"),
 450                    expected_value=date.minutes,
 451                    as_type=int,
 452                    failure_msg="Failed to set the minutes for the end time.")
 453            if date.am_pm and date.am_pm is not None:
 454                IAAssert.is_equal_to(
 455                    actual_value=Select(
 456                        webelement=self._end_time_am_pm_select.find()).first_selected_option.text.upper(),
 457                    expected_value=date.am_pm.upper(),
 458                    as_type=str,
 459                    failure_msg="Failed to set AM/PM for the end time.")

 461        def set_range(self, historical_range: HistoricalRange) -> None:
 462            """
 463            Set the Historical range by specifying all time pieces of both the beginning and end dates.
 464
 465            :param historical_range: A HistoricalRange object containing information about both the start date/time and
 466                end date/time for the desired range.
 467
 468            :raises AssertionError: If unsuccessful in applying any of the range values.
 469            """
 470            _beginning_hours = 0
 471            _ending_hours = 0
 472            if historical_range.beginning_hours is not None:
 473                _beginning_hours = int(historical_range.beginning_hours)
 474            if not historical_range.beginning_am:
 475                _beginning_hours += 12
 476            if historical_range.ending_hours is not None:
 477                _ending_hours = int(historical_range.ending_hours)
 478            if not historical_range.ending_am:
 479                _ending_hours += 12
 480            beginning_date = PerspectiveDate(
 481                year=historical_range.beginning_year,
 482                month=historical_range.beginning_month,
 483                day=historical_range.beginning_day,
 484                hours24=_beginning_hours,
 485                minutes=historical_range.beginning_minutes)
 486            ending_date = PerspectiveDate(
 487                year=historical_range.ending_year,
 488                month=historical_range.ending_month,
 489                day=historical_range.ending_day,
 490                hours24=_ending_hours,
 491                minutes=historical_range.ending_minutes)
 492            self.select_date_only_not_time(date=beginning_date)
 493            self.select_date_only_not_time(date=ending_date)
 494            self.set_start_time(date=beginning_date)
 495            self.set_end_time(date=ending_date)

The DateRange Selector used in multiple larger components, including the Alarm Journal Table, the Power Chart, and During filtering of Table columns which render as a date.

In some cases the range selector may change between a Historical 'Picker' which allows for specifying a beginning date/time and an ending date/time, or a Realtime range selector which allows for specifying a range of time before the current time, where a value of '8 Hours' would specify a range of anything between 'now and 8 hours ago'.

Use of the Historical picker requires the presence of the default chevrons/arrows/spinners.
