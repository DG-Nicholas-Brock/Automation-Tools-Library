144    def click_visible_base_item(self, text_of_base_item: str, binding_wait_time: float = 0) -> None:
145        """
146        Click a base-level item by text.
147
148        :raises NoSuchElementException: If unable to locate any top-level item with the supplied text.
149        """
150        while (not self.base_item_is_visible(
151                text_of_base_item=text_of_base_item)) \
152                and (self.is_compressed()) \
153                and self.can_scroll_right():
154            self.short_scroll(scroll_right=True)
155        while (not self.base_item_is_visible(
156                text_of_base_item=text_of_base_item)) \
157                and (self.is_compressed()) \
158                and self.can_scroll_left():
159            self.short_scroll(scroll_right=False)
160        if not self.base_item_is_visible(
161                text_of_base_item=text_of_base_item):
162            raise NoSuchElementException(f'No item with text of \'{text_of_base_item}\' could be found')
163        self._get_top_level_menu_item(
164            text_of_base_item=text_of_base_item).click(binding_wait_time=binding_wait_time)

240    def short_scroll(self, scroll_right: bool) -> None:
241        """
242        Scroll the Horizontal Menu (left/right) a small amount. Useful when searching for a desired top-level item.
243
244        :param scroll_right: If True, the Horizontal Menu will be scrolled right. If False, the Horizontal Menu

246        """
247        side_nav_to_click = self._side_scroll_clickable_area.find_all()[1 if scroll_right else 0]
248        IASelenium(driver=self.driver).long_click(web_element=side_nav_to_click)

250    def _get_child_menu_item(self, text_of_child_item: str) -> ComponentPiece:
251        """
252        Retrieve a child item Component, or construct a new one if it doesn't exist already.
253        """
254        item = self._child_item_coll.get(text_of_child_item)
255        if not item:
256            # Here, we're using the Component Modal as the direct parent WITHOUT including a reference to the actual
257            # component, even though the modal is typically a CHILD of the component. There are edge cases
258            # (fullscreen mode) where this is not always the case, so we can't force the longer locator.
259            item = ComponentPiece(
260                locator=(By.CSS_SELECTOR, f'.horizontal-menu-menu-item[data-label="{text_of_child_item}"]'),
261                driver=self.driver,
262                parent_locator_list=self._child_item_modal.locator_list,
263                wait_timeout=2,
264                poll_freq=self.poll_freq)
265            self._child_item_coll[text_of_child_item] = item
266        return item

268    def _get_top_level_menu_item(self, text_of_base_item: str) -> ComponentPiece:
269        """
270        Retrieve a top-level item Component, or construct a new one if it doesn't exist already.
271        """
272        item = self._menu_item_coll.get(text_of_base_item)
273        if not item:
274            item = ComponentPiece(
275                locator=(By.CSS_SELECTOR, f'{self._MENU_ITEM_LOCATOR[1]}[data-label="{text_of_base_item}"]'),
276                driver=self.driver,
277                parent_locator_list=self.locator_list,
278                wait_timeout=2,
279                poll_freq=self.poll_freq)
280            self._menu_item_coll[text_of_base_item] = item
281        return item

283    def _get_child_item_icon(self, text_of_child_item: str) -> CommonIcon:
284        """
285        Retrieve a child item ICON Component, or construct a new one if it doesn't exist already.
286        """
287        icon = self._top_level_menu_item_icon_coll.get(text_of_child_item)
288        if not icon:
289            icon = CommonIcon(
290                locator=(By.CSS_SELECTOR, 'svg'),
291                driver=self.driver,
292                parent_locator_list=self._get_child_menu_item(
293                    text_of_child_item=text_of_child_item).locator_list,
294                wait_timeout=1,
295                poll_freq=self.poll_freq)
296            self._top_level_menu_item_icon_coll[text_of_child_item] = icon
297        return icon
