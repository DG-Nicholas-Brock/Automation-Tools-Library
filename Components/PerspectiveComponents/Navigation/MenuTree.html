 57    def click_back_action_within_submenu(self, until_at_top_level: bool = False) -> None:
 58        """
 59        Click the 'Back' action available for the currently expanded submenu. If no submenu is expanded, no action is
 60        taken.
 61
 62        :raises TimeoutException: If no 'Back' action is currently available, or if no submenu is displayed.
 63        """
 64        if self.submenu_is_expanded():
 65            self._get_submenu_back_action().find_all(wait_timeout=0)[-1].click()
 66            self._get_submenu_back_action().wait_on_binding(0.5)  # allow submenu time to be removed from DOM
 67            try:
 68                if until_at_top_level and self._get_submenu_back_action().find(wait_timeout=0) is not None:
 69                    self.click_back_action_within_submenu(until_at_top_level=until_at_top_level)
 70            except TimeoutException:
 71                pass
 72        if until_at_top_level:
 73            IAAssert.is_not_true(
 74                value=self.submenu_is_expanded(),
 75                failure_msg="Failed to get the Menu Tree back to the top level.")

330    def wait_on_item_presence(
331            self,
332            item_text: str,
333            should_be_present: bool,
334            is_in_submenu: bool = False,
335            wait_timeout: Optional[float] = None):
336        """
337        Wait some period of time until an item either becomes present in the Menu Tree or is removed from the Menu Tree.
338
339        :param item_text: The text of the item to query for until it has the expected presence.
340        :param should_be_present: Specifies whether you are waiting for the item to be present or to not be present.
341        :param is_in_submenu: Clarifies whether the item should be expected as a top-level item or within a submenu.
342        :param wait_timeout: How long (in seconds) to remain polling for the item.
343
344        :return: True, if the item presence state matches the state specified in `should_be_present` - False otherwise.
345        """
346        item = self._get_item_or_submenu_item(item_text=item_text, is_in_submenu=is_in_submenu)
347        wait = self.wait if wait_timeout is None else WebDriverWait(driver=self.driver, timeout=wait_timeout)
348
349        def check_for_presence():
350            nonlocal item
351            nonlocal should_be_present
352            try:
353                return (item.find(wait_timeout=0) is not None) == should_be_present
354            except TimeoutException:
355                return should_be_present is False  # item wasn't found, which might be expected
356
357        try:
358            return wait.until(IAec.function_returns_true(custom_function=check_for_presence, function_args={}))
359        except TimeoutException:
360            return False
