250            """
251            Sets the Google Map to the type that matches the supplied subtype.
252
253            :param map_subtype: The subtype to set the Google Map to.
254            :param should_be_selected: Decides what selected state the subtype should be in.
255
256            :raises TimeoutException: If the MapType UI is not found or if this method is called for the dropdown menu
257                version.
258            """
259            if map_subtype.value == MapSubtype.TERRAIN.value:
260                self._map_control.click()
261                if self.terrain_checkbox_is_selected() != should_be_selected:
262                    self.wait.until(ec.presence_of_element_located(self._TERRAIN_LIST_ITEM_LOCATOR)).click()
263                    self.wait_on_binding()
264            else:
265                self._satellite_control.click()
266                if self.labels_checkbox_is_selected() != should_be_selected:
267                    self.wait.until(ec.presence_of_element_located(self._LABELS_LIST_ITEM_LOCATOR)).click()
268                    self.wait_on_binding()

498    def get_map_center(self) -> GeographicPoint:
499        """
500        Obtains the central latitude and longitude point of the Google Map's current position.
