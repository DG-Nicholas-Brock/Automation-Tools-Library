339    def drag_widget_to_column_row_and_release(
340            self,
341            one_based_column_index: int,
342            one_based_row_index: int,
343            widget: Widget) -> None:
344        """
345        Click within a widget and drag it to a specified cell, before then releasing the click.
346
347        :param one_based_column_index: The one-based index of the column we will drag the widget to.
348        :param one_based_row_index: The one-based index of the row we will drag the widget to.
349        :param widget: The Widget object which contains information about the widget we will be dragging.
350
351        :raises TimeoutException: If the Widget is not present, or if the Dashboard is not in editing mode, or if the
352            supplied cell indices are not valid.
353        """
354        cell_x_offset = int(float(self.get_cell_width())) // 2
355        cell_y_offset = int(float(self.get_cell_height())) // 2
356        # click-and-hold the widget in the center of the 1st (upper-left) cell, move the mouse to the center of the
357        # supplied cell, and then release the held click.
358        ActionChains(
359            driver=self.driver) \
360            .move_to_element_with_offset(
361            to_element=widget.get_widget(),
362            xoffset=cell_x_offset,
363            yoffset=cell_y_offset) \
364            .click_and_hold() \
365            .move_to_element_with_offset(
366            to_element=self._get_grid_cell(
367                one_based_column_index,
368                one_based_row_index).find(),
369            xoffset=int(float(self.get_cell_width())),
370            yoffset=int(float(self.get_cell_height()))) \
371            .release() \
372            .perform()
