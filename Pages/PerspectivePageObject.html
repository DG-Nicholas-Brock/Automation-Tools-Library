 54                self, item_text: str, submenu_depth: Optional[int] = None, binding_wait_time: float = 0.5) -> None:
 55            """
 56            Click an item within the context menu. For items within submenus, you must first hover over whichever other
 57            items would expand the submenus.
 58
 59            :param item_text: The text of the item to click.
 60            :param submenu_depth: The depth of the submenu to which the item belongs.
 61            :param binding_wait_time: The amount of time to wait after clicking the item before allowing code to
 62                continue.
 63            """
 64            if submenu_depth is not None:
 65                # For sub-menu items, we must first hover over the 0th element in that sub-menu, otherwise
 66                # the sub-menu collapses as we leave our current item
 67                self._get_item(submenu_depth=submenu_depth).hover()
 68            self._get_item(item_text=item_text, submenu_depth=submenu_depth).click(binding_wait_time=binding_wait_time)

216        def _get_expansion_icon(self, item_text: str, submenu_depth: Optional[int] = None) -> CommonIcon:
217            """
218            Obtain the expansion icon in use for an item of the context menu, based on the item's text and submenu
219                depth.
220
221            :param item_text: The text of the item from which you would like the expansion icon.
222            :param submenu_depth: The submenu depth of the expansion icon to be obtained. A None value implies the icon
223                is at the top-most level of the context menu, while 0 implies the icon is within the 0th nested submenu,
224                etc.
225
226            :return: The ComponentPiece which defines the expansion icon of the item with the supplied text at the
227                specified submenu depth.
228            """
229            identifier = (item_text, submenu_depth)
230            icon = self._icon_coll.get(identifier)
231            if not icon:
232                icon = CommonIcon(
233                    locator=self._SUBMENU_EXPANSION_ICON_LOCATOR,
234                    driver=self.driver,
235                    parent_locator_list=self._get_item(item_text=item_text, submenu_depth=submenu_depth).locator_list,
236                    wait_timeout=1)
237                self._icon_coll[identifier] = icon
238            return icon

240        def _get_icon(self, item_text: str, submenu_depth: Optional[int] = None) -> CommonIcon:
241            """
242            Obtain the icon in use for an item of the context menu, based on the item's text and submenu depth.
243
244            :param item_text: The text of the item from which you would like the icon.
245            :param submenu_depth: The submenu depth of the icon to be obtained. A None value implies the icon is
246                at the top-most level of the context menu, while 0 implies the icon is within the 0th nested submenu,
247                etc.
248
249            :return: The ComponentPiece which defines the icon of the item with the supplied text at the specified
250                submenu depth.
251            """
252            identifier = (item_text, submenu_depth)
253            icon = self._icon_coll.get(identifier)
254            if not icon:
255                parent_locator_list = self._get_item(item_text=item_text, submenu_depth=submenu_depth).locator_list
256                icon = CommonIcon(
257                    locator=(By.CSS_SELECTOR, 'svg'),
258                    driver=self.driver,
259                    parent_locator_list=parent_locator_list,
260                    wait_timeout=1)
261                self._icon_coll[identifier] = icon
262            return icon

264        def _get_item(self, item_text: Optional[str] = None, submenu_depth: Optional[int] = None) -> ComponentPiece:
265            """
266            Obtain the Component Piece defined by text and depth.
267
268            :param item_text: The text of the item you wish to obtain.
269            :param submenu_depth: The submenu depth of the item to be obtained. A None value implies the menu item is
270                at the top-most level of the context menu, while 0 implies the item is within the 0th nested submenu,
271                etc.
272
273            :return: The Component Piece defined by the supplied text and submenu depth.
274            """
275            identifier = (item_text, submenu_depth)
276            item = self._item_coll.get(identifier)
277            if not item:
278                parent_locator_list = self.locator_list if submenu_depth is None \
279                    else [self._get_submenu_locator(submenu_depth=submenu_depth)]
280                # if item_text is None, we're looking for the submenu wrapper, so ignore any label attribute
281                label_piece = f'[data-label="{item_text}"]' if item_text is not None else ""
282                item = ComponentPiece(
283                    locator=(By.CSS_SELECTOR, f'{self._ITEM_LOCATOR[1]}{label_piece}'),
284                    driver=self.driver,
285                    parent_locator_list=parent_locator_list,
286                    wait_timeout=1)
287                self._item_coll[identifier] = item
288            return item

266            """
267            Obtain the Component Piece defined by text and depth.
268
269            :param item_text: The text of the item you wish to obtain.
270            :param submenu_depth: The submenu depth of the item to be obtained. A None value implies the menu item is
271                at the top-most level of the context menu, while 0 implies the item is within the 0th nested submenu,
272                etc.
273
274            :return: The Component Piece defined by the supplied text and submenu depth.
275            """
276            identifier = (item_text, submenu_depth)
277            item = self._item_coll.get(identifier)
278            if not item:
279                parent_locator_list = self.locator_list if submenu_depth is None \
280                    else [self._get_submenu_locator(submenu_depth=submenu_depth)]
281                # if item_text is None, we're looking for the submenu wrapper, so ignore any label attribute
282                label_piece = f'[data-label="{item_text}"]' if item_text is not None else ""
283                item = ComponentPiece(
284                    locator=(By.CSS_SELECTOR, f'{self._ITEM_LOCATOR[1]}{label_piece}'),
285                    driver=self.driver,
286                    parent_locator_list=parent_locator_list,
287                    wait_timeout=1)
288                self._item_coll[identifier] = item
289            return item
