 93    def click_first_page_button(self) -> None:
 94        """
 95        Click the option in the Page which would take a user to the 'first' page.
 96
 97        :raises TimeoutException: If the 'First' page option is not present in the Pager.
 98        :raises AssertionError: If clicking the 'First' page button does result in the Table displaying the first page.
 99        """
100        self._first_page_button.click(binding_wait_time=0.5)
101        if len(self.get_all_listed_pages()) > 0:
102            IAAssert.is_equal_to(
103                actual_value=self.get_active_page(),
104                expected_value=1,
105                as_type=int,
106                failure_msg="We failed to get back to the first page of the Table.")

117    def click_next_page_chevron(self, binding_wait_time: float = 0.5) -> None:
118        """
119        Click the chevron which would take the user to the next page.
120
121        :param binding_wait_time: The amount of time (in seconds) after clicking the next page chevron before allowing
122            code to continue.
123
124        :raises TimeoutException: If the 'next' chevron does not exist, or if no page numbers exist in the Pager.
125        """
126        original_page_number = self.get_active_page()
127        self._next_page_chevron.click(binding_wait_time=binding_wait_time)
128        IAAssert.is_equal_to(
129            actual_value=self.get_active_page(),
130            expected_value=original_page_number + 1,
131            as_type=int,
132            failure_msg="We failed to get to the next page after clicking the next page chevron.")

134    def click_page_number(self, desired_page: Union[int, str]) -> None:
135        """
136        Click a page number within the Pager.
137
138        :param desired_page: The page number you wish to click.
139
140        :raises TimeoutException: If no page numbers are present in the Pager.
141        :raises IndexError: If the specified page number is not present in the Pager.
142        :raises AssertionError: If the Table does not end up displaying the specified page number.
143        """
144        list(filter(lambda e: e.text == str(desired_page), self._page_numbers.find_all()))[0].click()
145        IAAssert.is_equal_to(
146            actual_value=self.get_active_page(),
147            expected_value=desired_page,
148            as_type=int,
149            failure_msg=f"After clicking the option for page {desired_page}, we failed to display that page.")

258    def jump_to_page(self, page_to_jump_to: Union[int, str], binding_wait_time: float = 1) -> None:
259        """
260        Use the page jump input field to go to a specific page in the Table.
261
262        :param page_to_jump_to: The page of the Table to go to.
263        :param binding_wait_time: The amount of time (in seconds) to wait after supplying the page number before
264            allowing code to continue.
265
266        :raises TimeoutException: If the page jump input is not present.
267        :raises AssertionError: If the Table does not end up on the supplied page.
268        """
269        self._page_jump_input.find().click()
270        self.driver.execute_script('arguments[0].value = ""', self._page_jump_input.find())
271        self._page_jump_input.find().send_keys(str(page_to_jump_to) + Keys.ENTER)
272        self._page_jump_input.wait_on_binding(binding_wait_time)
273        # we can only manage the following assertion if page numbers are listed
274        if len(self.get_all_listed_pages()) > 0:
275            IAAssert.is_equal_to(
276                actual_value=self.get_active_page(),
277                expected_value=page_to_jump_to,
278                as_type=int,
279                failure_msg=f"Failed to jump to the specified page ({page_to_jump_to}).")

389    def set_displayed_row_count(self, count_of_rows: Union[int, str], attempted=False) -> None:
390        """
391        Set the number of displayed rows in the Table. The count of rows must be a valid value from the
392        dropdown (Select) in the Pager.
393
394        :param count_of_rows: The desired count of rows (this must be one of the available values in the dropdown).
395        :param attempted: Has the function been tried already, if False and counts do not match try to set the dropdown
396            value once more.
397
398        :raises TimeoutException: If the row count dropdown is not present.
399        :raises AssertionError: If unable to set the row count dropdown to the supplied row count.
400        """
401        Select(webelement=self._row_count_select.find()).select_by_value(str(count_of_rows))
402        self.wait_on_binding(time_to_wait=1)
403        if str(count_of_rows) != str(self.get_selected_row_count_option_from_dropdown()):
404            if not attempted:
405                self.set_displayed_row_count(count_of_rows, attempted=True)
406            else:
407                # already tried once...
408                pass
409        IAAssert.is_equal_to(
410            actual_value=self.get_selected_row_count_option_from_dropdown(),
411            expected_value=count_of_rows,
412            as_type=int,
413            failure_msg="We failed to specify a count of rows to select.")
