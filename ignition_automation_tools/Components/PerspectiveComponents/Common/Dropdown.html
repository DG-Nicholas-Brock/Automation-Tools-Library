105        """

199    def option_is_visible_within_option_modal(self, option_text: str) -> bool:
200        """
201        Determine if an option is visible within the modal which displays options.
202
203        :param option_text: The text of the option you wish to verify is contained within the modal.
204
205        :returns: True, if the supplied option is present and is rendered within the boundaries of the modal - False if
206            the option is not displayed at all, or if the option is rendering even partially outside the bounds of the
207            modal - including options which must be scrolled to.
208        """
209        modal = ComponentModal(driver=self.driver)
210        try:
211            visible = self._get_option(option_text=option_text).find().is_displayed()
212        except TimeoutException:  # option not even found
213            return False
214        if not visible:
215            # don't bother checking anything else
216            return False
217        option_origin = self._get_option(option_text=option_text).get_origin()
218        option_termination = self._get_option(option_text=option_text).get_termination()
219        modal_origin = modal.get_origin()
220        modal_termination = modal.get_termination()
221        left_within_modal = option_origin.X >= modal_origin.X
222        top_within_modal = option_origin.Y >= modal_origin.Y
223        right_within_modal = option_termination.X <= modal_termination.X
224        bottom_within_modal = option_termination.Y <= modal_termination.Y
225        return visible and left_within_modal and top_within_modal and right_within_modal and bottom_within_modal

227    def select_option_by_text_if_not_selected(self, option_text: str, binding_wait_time: float = 0.5) -> None:
228        """
229        Select an option from the Dropdown if it is not already selected. No action is taken if the supplied option is
230        already selected.
231
232        :param option_text: The exact text of the option to select.
233        :param binding_wait_time: The amount of time (in seconds) to wait after selecting the option before allowing
234            code to continue. Ignored if the option is already selected.
235
236        :raises TimeoutException: If the specified option is not present.
237        :raises AssertionError: If unsuccessful in selecting the specified option.
238        """
239        if option_text not in self.get_selected_options_as_list():
240            self.scroll_to_element()
241            self.expand()
242            try:
243                self._get_option(option_text=option_text).click(wait_timeout=1, binding_wait_time=binding_wait_time)
244            except TimeoutException as toe:
245                raise TimeoutException(msg=f"Failed to locate element with text of \"{option_text}\".") from toe
246            except ElementNotInteractableException as enie:
247                origin = self._options_container.get_origin()
248                termination = self._options_container.get_termination()
249                raise ElementNotInteractableException(
250                    msg=f"The container and options were found, but could not be clicked.\n"
251                        f"Container dimensions: {termination.X - origin.X} x {termination.Y - origin.Y}") from enie
252        assert option_text in self.get_selected_options_as_list(), f"Failed to select option: '{option_text}'."

254    def wait_for_expansion_state(self, state_to_wait_on: ExpansionState, wait_timeout: float = 1) -> bool:
255        """
256        Wait for the Dropdown to take on an expected state, before eventually returning a value which reflects whether
257        the Dropdown took that state.
258
259        :param state_to_wait_on: Either expanded or collapsed.
260        :param wait_timeout: The amount of time (in seconds) to potentially wait for the Dropdown to take on the
261            supplied state.
262
263        :return: True, if the Dropdown eventually took the supplied state - False otherwise.
264        """
265        exp_func = IAec.function_returns_true if state_to_wait_on == ExpansionState.EXPANDED \
266            else IAec.function_returns_false
267        try:
268            WebDriverWait(driver=self.driver, timeout=wait_timeout).until(
269                exp_func(custom_function=self.is_expanded, function_args={}))
270            return True
271        except TimeoutException:
272            return False

274    def _click_expansion_icon(self) -> None:
275        """
276        Expands the Dropdown if it is collapsed, or collapses the dropdown if it is expanded.
277
278        :raises AssertionError: If clicking the expansion icon is unsuccessful in changing the state of the Dropdown.
279        """
280        expected_final_state = ExpansionState.COLLAPSED if self.is_expanded() else ExpansionState.EXPANDED
281        self._expand_icon.click()
282        IAAssert.is_true(
283            value=self.wait_for_expansion_state(state_to_wait_on=expected_final_state, wait_timeout=1),
284            failure_msg="Failed to modify the expansion state of the Dropdown."
285        )

287    def _get_option(self, option_text: str) -> ComponentPiece:
288        """
289        Obtain the ComponentPiece which defines an option within the modal.
290
291        :param option_text: The exact text of the option.
292
293        :returns: A ComponentPiece which defines the option of the Dropdown as it will be located within the modal
294            during expansion.
295        """
296        option_element = self._option_elements.get(option_text)
297        if not option_element:
298            option_element = ComponentPiece(
299                locator=(By.CSS_SELECTOR, f'a[data-label="{option_text}"]'),
300                driver=self.driver,
301                parent_locator_list=self._options_container.locator_list,
302                poll_freq=self.poll_freq)
303            self._option_elements[option_text] = option_element
304        return option_element
