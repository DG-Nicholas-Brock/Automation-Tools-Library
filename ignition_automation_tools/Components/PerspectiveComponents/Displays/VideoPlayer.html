290    def drag_seek_handle_to_time(self, time_as_hour_minute_second_string: str) -> None:
291        """
292        Drag the seek handle (scrubber) to a specific colon-delimited time. Currently only supports changes in seconds.
293        This function takes longer the larger the Video Player is because it drags one pixel at a time.
294
295        :raises ElementClickInterceptedException: If the base controls are configured to hide during times of
296            inactivity.
297        """
298        desired_seconds = time_as_hour_minute_second_string.split(':')[-1]
299        current_seconds = self.get_elapsed_time().split(':')[-1]
300        directional_modifier = 1 if desired_seconds > current_seconds else -1
301        value_needs_adjustment = desired_seconds != current_seconds
302        if value_needs_adjustment:
303            ActionChains(driver=self.driver).click_and_hold(on_element=self._seek_handle.find()).perform()
304        if desired_seconds < current_seconds:
305            while desired_seconds < current_seconds:
306                ActionChains(driver=self.driver).move_by_offset(xoffset=directional_modifier, yoffset=0).perform()
307                current_seconds = self.get_elapsed_time().split(':')[-1]
308        elif desired_seconds > current_seconds:
309            while desired_seconds > current_seconds:
310                ActionChains(driver=self.driver).move_by_offset(xoffset=directional_modifier, yoffset=0).perform()
311                current_seconds = self.get_elapsed_time().split(':')[-1]
312        if value_needs_adjustment:
313            ActionChains(driver=self.driver).release().perform()

424    def set_volume(self, desired_volume: Union[int, str]) -> None:
425        """
426        Set the volume of the Video Player.
427
428        :param desired_volume: The value (0<=desired_value<=100) to which the volume will be set.
429
430        :raises ElementClickInterceptedException: If the base controls are configured to hide during times of
431            inactivity.
432        :raises ValueError: If the value is invalid.
433        """
434        desired_volume = int(desired_volume)
435        if (0 > desired_volume) or (desired_volume > 100):
436            raise ValueError("Volume only accepts values between 0 and 100.")
437        if desired_volume != self.get_current_volume():
438            self._set_volume(desired_volume)

444    def _select_play_rate(self, desired_rate: PlayRate) -> None:
445        """Set the play rate of the Video Player."""
446        options = self._play_rate_options.find_all()
447        for option in options:
448            if option.text == desired_rate.value:
449                option.click()

452    def _set_volume(self, desired_volume: int) -> None:
453        """Set the volume of the Video Player."""
454        desired_volume = int(desired_volume)
455        current_volume = self.get_current_volume()
456        directional_modifier = 1 if current_volume > desired_volume else -1
457        value_needs_adjustment = current_volume != desired_volume
458        if value_needs_adjustment:
459            self.click_volume_icon()
460            ActionChains(driver=self.driver).click_and_hold(on_element=self._volume_slider_handle.find()).perform()
461        while current_volume != desired_volume:
462            ActionChains(driver=self.driver).move_by_offset(xoffset=0, yoffset=directional_modifier).perform()
463            current_volume = int(self._volume_percent.find(
464                wait_timeout=1).get_attribute("style").split(' ')[-1].split('%')[0])
465        if value_needs_adjustment:
466            ActionChains(driver=self.driver).release().perform()
467        if self._volume_modal_is_displayed():
468            self.click_volume_icon()

471        """Determine if the volume modal is displayed."""

477    @staticmethod
478    def _convert_time_to_seconds(time_as_str: str) -> int:
479        """Convert a colon-delimited time string to a count of seconds."""
480        if len(time_as_str.split(":")) == 2:
481            # ensure the presence of hours in the string
482            time_as_str = "00:" + time_as_str
483        ftr = [3600, 60, 1]  # seconds in hour, seconds in minute, seconds in second
484        return sum([a * b for a, b in zip(ftr, map(int, time_as_str.split(':')))])

291    def drag_seek_handle_to_time(self, time_as_hour_minute_second_string: str) -> None:
292        """
293        Drag the seek handle (scrubber) to a specific colon-delimited time. Currently only supports changes in seconds.
294        This function takes longer the larger the Video Player is because it drags one pixel at a time.
295
296        :raises ElementClickInterceptedException: If the base controls are configured to hide during times of
297            inactivity.
298        """
299        desired_seconds = time_as_hour_minute_second_string.split(':')[-1]
300        current_seconds = self.get_elapsed_time().split(':')[-1]
301        directional_modifier = 1 if desired_seconds > current_seconds else -1
302        value_needs_adjustment = desired_seconds != current_seconds
303        if value_needs_adjustment:
304            ActionChains(driver=self.driver).click_and_hold(on_element=self._seek_handle.find()).perform()
305        if desired_seconds < current_seconds:
306            while desired_seconds < current_seconds:
307                ActionChains(driver=self.driver).move_by_offset(xoffset=directional_modifier, yoffset=0).perform()
308                current_seconds = self.get_elapsed_time().split(':')[-1]
309        elif desired_seconds > current_seconds:
310            while desired_seconds > current_seconds:
311                ActionChains(driver=self.driver).move_by_offset(xoffset=directional_modifier, yoffset=0).perform()
312                current_seconds = self.get_elapsed_time().split(':')[-1]
313        if value_needs_adjustment:
314            ActionChains(driver=self.driver).release().perform()

478    @staticmethod
479    def _convert_time_to_seconds(time_as_str: str) -> int:
480        """Convert a colon-delimited time string to a count of seconds."""
481        if len(time_as_str.split(":")) == 2:
482            # ensure the presence of hours in the string
483            time_as_str = "00:" + time_as_str
484        ftr = [3600, 60, 1]  # seconds in hour, seconds in minute, seconds in second
485        return sum([a * b for a, b in zip(ftr, map(int, time_as_str.split(':')))])

def drag_seek_handle_to_time(self, time_as_hour_minute_second_string: str) -> None: 
291    def drag_seek_handle_to_time(self, time_as_hour_minute_second_string: str) -> None:
292        """
293        Drag the seek handle (scrubber) to a specific colon-delimited time. Currently only supports changes in seconds.
294        This function takes longer the larger the Video Player is because it drags one pixel at a time.
295
296        :raises ElementClickInterceptedException: If the base controls are configured to hide during times of
297            inactivity.
298        """
299        desired_seconds = time_as_hour_minute_second_string.split(':')[-1]
300        current_seconds = self.get_elapsed_time().split(':')[-1]
301        directional_modifier = 1 if desired_seconds > current_seconds else -1
302        value_needs_adjustment = desired_seconds != current_seconds
303        if value_needs_adjustment:
304            ActionChains(driver=self.driver).click_and_hold(on_element=self._seek_handle.find()).perform()
305        if desired_seconds < current_seconds:
306            while desired_seconds < current_seconds:
307                ActionChains(driver=self.driver).move_by_offset(xoffset=directional_modifier, yoffset=0).perform()
308                current_seconds = self.get_elapsed_time().split(':')[-1]
309        elif desired_seconds > current_seconds:
310            while desired_seconds > current_seconds:
311                ActionChains(driver=self.driver).move_by_offset(xoffset=directional_modifier, yoffset=0).perform()
312                current_seconds = self.get_elapsed_time().split(':')[-1]
313        if value_needs_adjustment:
314            ActionChains(driver=self.driver).release().perform()
