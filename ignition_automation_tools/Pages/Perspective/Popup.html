 95    def click_modal_overlay(self) -> None:
 96        """
 97        Click the modal overlay. We accomplish this by hovering over the title bar, and then clicking slightly above
 98        the title bar.
 99
100        Clicking the overlay is a bit tricky because of browser differences; Chrome might click in the top-left, and
101        Firefox might click in the middle - where your Popup is likely to intercept the click. To get around this, we
102        actually target the title bar (if present) or Popup, then specify a location some number of pixels directly
103        above the Popup, blindly clicking at the location instead of targeting the modal itself.
104
105        :raises AssertionError: If the Popup is not a modal, or the overlay is not present.
106        """
107        IAAssert.is_true(
108            value=self.is_modal(),
109            failure_msg=f"Either this popup ({self.formatted_id}) is not a modal, or the overlay is not present.")
110        if self.has_title_bar():
111            # Targeting the title bar when available reduces the vertical offset, which reduces potential issues.
112            self.title_bar.hover()
113            height = self.title_bar.find().size.get('height')
114        else:
115            self.hover()
116            height = self.find().size.get('height')
117        y_offset = (height / 2 * -1) - 20  # safety handling for firefox vs chrome click location discrepancy
118        self._selenium.click_at_offset(y=y_offset)

121        """
122        Drag the Popup some direction by clicking the drag handle in the title bar.

125            left.
126        :param y_offset: The up/down value to drag the Popup. Positive values are DOWN, negative values are UP.

128        :raises TimeoutException: If the Popup is not present, or if the Popup is not configured to be draggable.
129        """
130        self._selenium.click_drag_release(web_element=self._drag_handle.find(), x=x_offset, y=y_offset)

273                description=f"The resize zone of the {self.formatted_id} Popup in the {resize_zone.value} region "
274                            f"({resize_zone}).",
275                wait_timeout=0)

110            failure_msg=f"Either this popup ({self.formatted_id}) is not a modal, or the overlay is not present.")
111        if self.has_title_bar():
112            # Targeting the title bar when available reduces the vertical offset, which reduces potential issues.
113            self.title_bar.hover()
114            height = self.title_bar.find().size.get('height')
115        else:
116            self.hover()
117            height = self.find().size.get('height')
118        y_offset = (height / 2 * -1) - 20  # safety handling for firefox vs chrome click location discrepancy
119        self._selenium.click_at_offset(y=y_offset)
