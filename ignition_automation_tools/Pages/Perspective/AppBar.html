 284    def about_modal_is_displayed(self) -> bool:
 285        """
 286        Determine if the 'About' panel is displayed.
 287
 288        :returns: True, if the 'About Ignition' panel is displayed - False otherwise.
 289        """
 290        try:
 291            return self._about_panel.find(wait_timeout=0).is_displayed() \
 292                and "open" in self._about_panel.find().get_attribute("class") \
 293                and self._about_modal_and_session_status_modal_title.get_text() != self._SESSION_STATUS_MODAL_TITLE_TEXT
 294        except TimeoutException:
 295            return False

 297    def action_panel_is_expanded(self) -> bool:
 298        """
 299        Determine if the Action Panel is currently displayed.
 300
 301        :returns: True, if the Action Panel is currently displayed - False otherwise.
 302        """
 303        try:

 305                self._action_panel.get_termination().Y <= self._app_bar_wrapper.get_origin().Y

 418    def collapse(self) -> None:
 419        """
 420        Collapse every piece of the App Bar, until only the Revealer is present.
 421
 422        :raises AssertionError: If unsuccessful in closing or collapsing every piece of the App Bar.
 423        """
 424        self.close_session_status_modal()
 425        self.close_about_modal_via_button()
 426        self.collapse_action_panel()
 427        if self.is_expanded():
 428            self._collapse_icon.click()
 429            self._wait_for_app_bar_to_be_fully_collapsed()
 430        IAAssert.is_not_true(
 431            value=self.is_expanded(),
 432            failure_msg="Failed to collapse the App Bar.")
