 44    def is_equal_to(
 45            actual_value,
 46            expected_value,
 47            failure_msg: Optional[str] = None,
 48            as_type=None) -> None:
 49        """
 50        Verify some value is equal to another value. When comparing two values which might be of different types, use
 51        the as_type argument to specify the type both values should be cast as; this is useful for ignoring instances
 52        where numeric and string values appear the same but differ in type.
 53
 54        :param actual_value: The actual value you are comparing/verifying.
 55        :param expected_value: The value expected to be reflected in the actual value.
 56        :param failure_msg: Any custom failure message you'd like to supply to explain the impact or cause of a
 57            potential assertion failure.
 58        :param as_type: Used to compare the two values as a specific type.
 59
 60        :raises AssertionError: If the supplied expected_value is an element of the supplied iterable.
 61        """
 62        if as_type is not None:
 63            if as_type == int:
 64                # must convert to float first!
 65                actual_value = float(actual_value)
 66                expected_value = float(expected_value)
 67            if as_type == Color:
 68                # Color only accepts a tuple constructor, which throws a wrench into our previous casting.
 69                assert IAAssert._color_convert(color_as_string=actual_value) ==\
 70                       IAAssert._color_convert(color_as_string=expected_value),\
 71                       IAAssert._generate_failure_msg(
 72                           msg=f'Assert {IAAssert._color_convert(color_as_string=actual_value)} == '
 73                               f'{IAAssert._color_convert(color_as_string=expected_value)} -> '
 74                               f'These values were converted to hex before comparison.',
 75                           failure_msg=failure_msg)
 76            elif as_type in [Tag, UdtDef, UdtInstance, Folder]:
 77                # Tags need to be compared as their dictionary counterparts.
 78                # msg will provide additional insight into any failure.
 79                assert actual_value.to_dict() == expected_value.to_dict(), \
 80                    IAAssert._generate_failure_msg(
 81                        msg=f'Dictionary Values do not match.  The following items are different:\n'
 82                            f'keys & values in Expected not in Actual: '
 83                            f'{expected_value.diff_my_unmatched_keys_values(actual_value)}\n'
 84                            f'keys & values in Actual not in Expected: '
 85                            f'{expected_value.diff_their_unmatched_keys_values(actual_value)}\n'
 86                            f'values that do not match: '
 87                            f'[{actual_value.diff_key_values(expected_tag=expected_value)}]',
 88                        failure_msg=failure_msg
 89                )
 90            else:
 91                assert as_type(actual_value) == as_type(expected_value), \
 92                    IAAssert._generate_failure_msg(msg=f'Assert {as_type(actual_value)} == '
 93                                                       f'{as_type(expected_value)}',
 94                                                   failure_msg=failure_msg)
 95        else:
 96            assert actual_value == expected_value, \
 97                IAAssert._generate_failure_msg(msg=f'Assert {str(type(actual_value))} {actual_value} == '
 98                                                   f'{str(type(expected_value))} {expected_value}',
 99                                               failure_msg=failure_msg)

102    def is_close_to(
103            actual_value: float,
104            expected_value: float,
105            decimal_places: int = 9,
106            failure_msg: Optional[str] = None):
107        """
108        Verify that two floating point values are close to one another, within a specified number of decimal places.
109        This is useful for verifying approximate equality while accounting for possible floating point imprecision.
110
111        :param actual_value: The actual value to be compared.
112        :param expected_value: The value that actual_value is expected to be close to.
113        :param decimal_places: The relative difference between the two values that we are willing to tolerate when
114            comparing them, in terms of decimal places (e.g. an argument of 3 asserts that the values are within 0.001
115            of each other)
116        :param failure_msg: Any custom failure message you'd like to supply to explain the impact or cause of a
117            potential assertion failure.
118
119        :raises AssertionError: If the provided values are not within the specified number of decimal places of one
120            another.
121        """
122        absolute_tolerance = abs(1 / (pow(10, decimal_places)))
123        assert math.isclose(actual_value, expected_value, abs_tol=absolute_tolerance, rel_tol=0),  \
124            IAAssert._generate_failure_msg(
125                msg=f'Assert {actual_value} is within {decimal_places} decimal places of {expected_value}',
126                failure_msg=failure_msg)

226            actual_value,
227            expected_value,
228            failure_msg: Optional[str] = None,
229            as_type=None) -> None:
230        """
231        Assert that two values are not equal. Due to type differences, this function should almost always have a
232        supplied type as part of the as_type argument.
233
234        Example: A value of numeric 2 compared to a value of string "2" will return True because they are different
235        types. Supplying a value of int or str for as_type will cast both as the supplied type resulting in supplied
236        values of 2 and "2" as returning False (it is not true that they are not equal, meaning they are equal)
237        because we will have removed the type difference as a root cause of the inequality.
238
239        :param actual_value: The actual value to be verified.
240        :param expected_value: The value expected to be reflected by the actual value.
241        :param failure_msg: Any custom failure message you'd like to supply to explain the impact or cause of a
242            potential assertion failure.
243        :param as_type: Used to compare the two values as a specific type.
244
245        :raises AssertionError: If the two values are equal.
246        """
247        if as_type is not None:
248            if as_type == int:
249                # must convert to float first!
250                actual_value = float(actual_value)
251                expected_value = float(expected_value)
252            if as_type == Color:
253                # special handling for color comparisons
254                assert IAAssert._color_convert(color_as_string=actual_value) \
255                       != IAAssert._color_convert(color_as_string=expected_value),\
256                       IAAssert._generate_failure_msg(
257                           msg=f'Assert {IAAssert._color_convert(color_as_string=actual_value)} != '
258                               f'{IAAssert._color_convert(color_as_string=expected_value)}.',
259                           failure_msg=failure_msg)
260            else:
261                assert as_type(actual_value) != as_type(expected_value),\
262                    IAAssert._generate_failure_msg(msg=f'Assert {as_type(actual_value)} != '
263                                                       f'{as_type(expected_value)}',
264                                                   failure_msg=failure_msg)
265        else:
266            assert actual_value != expected_value,\
267                IAAssert._generate_failure_msg(msg=f'Assert {actual_value} != {expected_value}',
268                                               failure_msg=failure_msg)

307    def is_within_range(
308            low_end_value: Union[int, float],
309            value: Union[int, float],
310            high_end_value: Union[int, float],
311            inclusive: bool,
312            failure_msg: str = None) -> None:
313        """
314        Assert some value falls within some defined range. Inclusion or exclusion is controlled by the inclusive arg.
315        For one-sided ranges, use less-than or greater-than assertion functions.
316
317        :param low_end_value: The low-end of your range.
318        :param value: The value you are verifying falls within the range.
319        :param high_end_value: The high-end of your range.
320        :param inclusive: Controls whether the low-end AND high-end of your range are inclusive.
321        :param failure_msg: Any custom failure message you'd like to supply to explain the impact or cause of a
322            potential assertion failure.
323
324        :raises AssertionError: If the supplied value does not fall within the defined range.
325        """
326        msg = IAAssert._generate_failure_msg(msg=f"Assert {low_end_value} <{'=' if inclusive else ''} {value}"
327                                                 f" <{'=' if inclusive else ''} {high_end_value}",
328                                             failure_msg=failure_msg)
329        if inclusive:
330            assert low_end_value <= value <= high_end_value, msg
331        else:
332            assert low_end_value < value < high_end_value, msg

334    @classmethod
335    def _color_convert(cls, color_as_string: str) -> str:
336        """
337        Selenium's color helper does not recognize alpha values in hex strings longer than 7 characters (#XXXXXX), so
338        if a hex value is supplied which is longer than seven characters we will force the kwarg conversion.
339        :param color_as_string: Your color as an rgb/rgba or hex string.
340        :returns: The string converted to a hex representation of a color
341        """
342        if color_as_string and color_as_string[0] == "#" and len(color_as_string) > 7 and color_as_string[-1] == "0":
343            _conv = Color.from_string(color_as_string)
344            return Color(red=_conv.red,
345                         green=_conv.green,
346                         blue=_conv.blue,
347                         alpha=0).hex
348        else:
349            return Color.from_string(color_as_string).hex
